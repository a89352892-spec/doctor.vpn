-- Функция создания кнопки
local function createButton(text, position, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1, 0, 0, 35)
    btn.Position = UDim2.new(0, 0, position, 0)
    btn.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
    btn.Text = text
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)
    btn.TextScaled = true
    btn.Font = Enum.Font.Gotham
    btn.Parent = buttonContainer
    
    btn.MouseButton1Click:Connect(callback)
    return btn
end

-- 1. Умный авто-сбор ресурсов
local collectRadius = 30
createButton("🌿 Авто-сбор ресурсов", 0, function()
    autoCollect = not autoCollect
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "🌲 ForestHunter",
        Text = "Авто-сбор " .. (autoCollect and "включен" or "выключен"),
        Duration = 2
    })
    
    while autoCollect do
        wait(0.5)
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("Model") or v:IsA("Part") then
                local name = v.Name:lower()
                -- Список ресурсов для сбора
                local resources = {"berry", "mushroom", "herb", "stone", "ore", "wood", "branch", "flower", "grass", "fruit"}
                for _, resource in pairs(resources) do
                    if name:find(resource) then
                        local distance = (rootPart.Position - v:GetPivot().Position).Magnitude
                        if distance < collectRadius then
                            rootPart.CFrame = v:GetPivot() + Vector3.new(0, 2, 0)
                            wait(0.3)
                            -- Имитация сбора
                            local tool = player.Backpack:FindFirstChildWhichIsA("Tool")
                            if tool then
                                tool.Parent = character
                                tool:Activate()
                                wait(0.3)
                                tool.Parent = player.Backpack
                            end
                        end
                    end
                end
            end
        end
    end
end)

-- 2. Авто-атака
local attackRange = 20
createButton("⚔️ Авто-атака", 40, function()
    autoKill = not autoKill
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "🌲 ForestHunter",
        Text = "Авто-атака " .. (autoKill and "включена" or "выключена"),
        Duration = 2
    })
    
    while autoKill do
        wait(0.1)
        local nearestEnemy = nil
        local nearestDist = attackRange
        
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("Model") and v:FindFirstChild("Humanoid") then
                -- Проверка что это враг (не игрок)
                if not game.Players:GetPlayerFromCharacter(v) then
                    local distance = (rootPart.Position - v:GetPivot().Position).Magnitude
                    if distance < nearestDist then
                        nearestEnemy = v
                        nearestDist = distance
                    end
                end
            end
        end
        
        if nearestEnemy then
            rootPart.CFrame = nearestEnemy:GetPivot() + Vector3.new(0, 0, 3)
            wait(0.1)
            -- Атака
            local tool = character:FindFirstChildWhichIsA("Tool")
            if tool then
                tool:Activate()
            else
                -- Если нет оружия, бьем руками
                humanoid:TakeDamage(10)
            end
        end
    end
end)

-- 5. Невидимость для мобов
createButton("👻 Невидимость (для мобов)", 200, function()
    invisibility = not invisibility
    
    if invisibility then
        -- Делаем персонажа невидимым
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Transparency = 1
            end
        end
        -- Отключаем коллизию
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
        -- Скрываем имя
        if character:FindFirstChild("Head") then
            character.Head:FindFirstChild("BillboardGui").Enabled = false
        end
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "🌲 ForestHunter",
            Text = "Невидимость включена! Мобы не видят тебя.",
            Duration = 3
        })
    else
        -- Возвращаем видимость
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Transparency = 0
                part.CanCollide = true
            end
        end
        if character:FindFirstChild("Head") and character.Head:FindFirstChild("BillboardGui") then
            character.Head.BillboardGui.Enabled = true
        end
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "🌲 ForestHunter",
            Text = "Невидимость выключена",
            Duration = 2
        })
    end
end)

-- 7. Name Tags (показывать имена и HP игроков)
createButton("🏷️ Name Tags (имена + HP)", 280, function()
    nameTags = not nameTags
    
    if nameTags then
        for _, p in pairs(game.Players:GetPlayers()) do
            if p ~= player then

                if char and char:FindFirstChild("Head") then
                    -- Создаем билборд с HP
                    local billboard = Instance.new("BillboardGui")
                    billboard.Name = "NameTagESP"
                    billboard.Size = UDim2.new(0, 200, 0, 50)
                    billboard.AlwaysOnTop = true
                    billboard.Parent = char.Head
                    
                    local nameLabel = Instance.new("TextLabel")
                    nameLabel.Size = UDim2.new(1, 0, 0.5, 0)
                    nameLabel.BackgroundTransparency = 1
                    nameLabel.Text = p.Name
                    nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
                    nameLabel.TextScaled = true
                    nameLabel.Font = Enum.Font.GothamBold
                    nameLabel.Parent = billboard
                    
                    local hpBar = Instance.new("Frame")
                    hpBar.Size = UDim2.new(1, 0, 0.4, 0)
                    hpBar.Position = UDim2.new(0, 0, 0.5, 0)
                    hpBar.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
                    hpBar.Parent = billboard
                    
                    local hpFill = Instance.new("Frame")
                    hpFill.Size = UDim2.new(1, 0, 1, 0)
                    hpFill.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
                    hpFill.Parent = hpBar
                    
                    -- Обновление HP
                    game:GetService("RunService").Heartbeat:Connect(function()
                        if char and char:FindFirstChild("Humanoid") then
                            local hp = char.Humanoid.Health / char.Humanoid.MaxHealth
                            hpFill.Size = UDim2.new(hp, 0, 1, 0)
                            hpFill.BackgroundColor3 = Color3.fromRGB(255 * (1 - hp), 255 * hp, 0)
                        end
                    end)
                end
            end
        end
    else
        -- Удаляем все NameTags
        for _, v in pairs(workspace:GetDescendants()) do
            if v.Name == "NameTagESP" then
                v:Destroy()
            end
        end
    end
end)

-- 8. Item ESP (подсветка ценных предметов)
createButton("💎 Item ESP (ценные предметы)", 320, function()
    itemESP = not itemESP
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "🌲 ForestHunter",
        Text = "Item ESP " .. (itemESP and "включен" or "выключен"),
        Duration = 2
    })
    
    if itemESP then
        -- Создаем ESP для предметов
        local function addESP(item)
            if item:IsA("Model") or item:IsA("BasePart") then
                local name = item.Name:lower()
                local valuable = {"chest", "gold", "diamond", "crystal", "epic", "legendary", "rare", "key", "boss", "treasure"}
                
                for _, v in pairs(valuable) do
                    if name:find(v) then
                        local highlight = Instance.new("Highlight")
                        highlight.Adornee = item
                        highlight.FillColor = Color3.fromRGB(255, 215, 0) -- Золотой
                        highlight.FillTransparency = 0.3
                        highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
                        highlight.Name = "ItemESP"
                        highlight.Parent = item
                        
                        -- Добавляем название
                        local billboard = Instance.new("BillboardGui")
                        billboard.Name = "ItemNameESP"
                        billboard.Size = UDim2.new(0, 150, 0, 30)
                        billboard.AlwaysOnTop = true
                        billboard.Parent = item
                        
                        local text = Instance.new("TextLabel")
                        text.Size = UDim2.new(1, 0, 1, 0)
                        text.BackgroundTransparency = 1
                        text.Text = item.Name
                        text.TextColor3 = Color3.fromRGB(255, 215, 0)
                        text.TextScaled = true
                        text.Font = Enum.Font.GothamBold
                        text.Parent = billboard
                        break
                    end
                end
            end
        end
        
        -- Сканируем мир
        for _, v in pairs(workspace:GetDescendants()) do
            addESP(v)
        end
        
        -- Сканируем новые предметы
        workspace.DescendantAdded:Connect(addESP)
    else
        -- Удаляем все ESP
        for _, v in pairs(workspace:GetDescendants()) do
            if v.Name == "ItemESP" or v.Name == "ItemNameESP" then
                v:Destroy()
            end
        end
    end
end)

createButton("🔥 Авто-заправка костра (до 6 ур.)", 360, function()
    autoCampfire = not autoCampfire
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "🌲 ForestHunter",
        Text = "Авто-заправка " .. (autoCampfire and "включена" or "выключена"),
        Duration = 2
    })
    
    while autoCampfire do
        wait(1)
        for _, v in pairs(workspace:GetDescendants()) do
            if v.Name == "Campfire" or v.Name == "Fire" then
                -- Ищем инвентарь для заправки
                local inventory = player.Backpack:GetChildren()
                for _, item in pairs(inventory) do
                    if item:IsA("Tool") and (item.Name:lower():find("wood") or item.Name:lower():find("log") or item.Name:lower():find("fuel")) then
                        -- Телепортируемся к костру
                        rootPart.CFrame = v.CFrame + Vector3.new(0, 2, 2)
                        wait(0.5)
                        -- Используем предмет
                        item.Parent = character
                        item:Activate()
                        wait(1)
                        item.Parent = player.Backpack
                        -- Проверяем уровень
                        if v:FindFirstChild("Level") and v.Level.Value >= 6 then
                            game:GetService("StarterGui"):SetCore("SendNotification", {
                                Title = "🌲 ForestHunter",
                                Text = "🔥 Костер достиг 6 уровня!",
                                Duration = 3
                            })
                            autoCampfire = false
                        end
                        break
                    end
                end
                break
            end
        end
    end
end)

-- Полет
createButton("✈️ Режим полета", 520, function()
    flyEnabled = not flyEnabled
    
    if flyEnabled then
        local fly = Instance.new("BodyVelocity")
        fly.Name = "FlyVelocity"
        fly.MaxForce = Vector3.new(4000, 4000, 4000)
        fly.Velocity = Vector3.new(0, 10, 0)
        fly.Parent = rootPart
        
        local userInput = game:GetService("UserInputService")
        local move = Vector3.new(0, 0, 0)
        
        userInput.InputBegan:Connect(function(input, gameProcessed)
            if gameProcessed then return end
            if input.KeyCode == Enum.KeyCode.W then move = move + Vector3.new(0, 0, -1) end
            if input.KeyCode == Enum.KeyCode.S then move = move + Vector3.new(0, 0, 1) end
            if input.KeyCode == Enum.KeyCode.A then move = move + Vector3.new(-1, 0, 0) end
            if input.KeyCode == Enum.KeyCode.D then move = move + Vector3.new(1, 0, 0) end
            if input.KeyCode == Enum.KeyCode.Space then move = move + Vector3.new(0, 1, 0) end
            if input.KeyCode == Enum.KeyCode.LeftShift then move = move + Vector3.new(0, -1, 0) end
        end)
        
        game:GetService("RunService").Heartbeat:Connect(function()
            if flyEnabled and rootPart:FindFirstChild("FlyVelocity") then
                rootPart.FlyVelocity.Velocity = move * 50
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "🌲 ForestHunter",
            Text = "Полет включен! (WASD + Space/Shift)",
            Duration = 3
        })
    else
        if rootPart:FindFirstChild("FlyVelocity") then
            rootPart.FlyVelocity:Destroy()
        end
    end
end)

-- Перетаскивание окна
local drag = Instance.new("TextButton")
drag.Size = UDim2.new(1, 0, 0, 45)
drag.BackgroundTransparency = 1
drag.Text = ""
drag.Parent = mainFrame

local dragging, dragInput, dragStart, startPos

drag.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

drag.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)
